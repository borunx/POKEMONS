//mostrar logo

public static void mostrarLogo() throws FileNotFoundException{
        //lanzamos la excepcion para que sea capturada en el main
        File logo_pokemon = new File("logo.pok.txt");
        Scanner lectura = new Scanner (logo_pokemon);

        while(lectura.hasNextLine()) {
            String logo = lectura.nextLine();
            System.out.println(logo);
        }
        lectura.close();
}
